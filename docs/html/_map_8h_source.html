346inline bool Map::addRoad(RoadInfo info){
347 auto v1 = findCity(info.source->getInfo());
348 auto v2 = findCity(info.dest->getInfo());
349 if (v1 == nullptr || v2 == nullptr)
350 return false;
351 auto e1 = v1->addRoad(info);
352 auto e2 = v2->addRoad({info.dest,info.source,info.drivingWeight,info.walkingWeight,info.drivable,info.walkable});
353 e1->setReverse(e2);
354 e2->setReverse(e1);
355 roads.push_back(e1);
356 roads.push_back(e2);
357 return true;
358}
