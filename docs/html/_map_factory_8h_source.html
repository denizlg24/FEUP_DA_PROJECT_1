77inline vector<RoadInfo> MapFactory::readRoads(const string& filename) const {
78 CSVParser csv_parser;
79 if (csv_parser.loadCSV(filename)) {
80 vector<RoadInfo> road_infos;
81 for (const auto& line: csv_parser.getData()) {
82 if (line.size() == 4) {
83 RoadInfo newRoad{};
84 string noKey = "X";
85 newRoad.source = getMap()->getCity(line[0]);
86 newRoad.dest = getMap()->getCity(line[1]);
87 newRoad.drivingWeight = line[2] == noKey ? 0 : stoi(line[2]);
88 newRoad.walkingWeight = line[3] == noKey ? 0 : stoi(line[3]);
89 newRoad.drivable = line[2] != noKey;
90 newRoad.walkable = line[3] != noKey;
91 road_infos.push_back(newRoad);
92 }
93 }
94 return road_infos;
95 } else {
96 __throw_runtime_error("Error opening file.");
